        function showPage(page) {

        function renderInventory() {

        }

        function createProductCard(product) {

            const card = document.createElement('div');

            card.className = 'toy-card bg-slate-950 border border-slate-700 rounded-3xl overflow-hidden cursor-pointer';

            let statusHTML = '';

            if (product.status === "In Stock") statusHTML = `<span class="status-badge bg-emerald-500 text-black">IN STOCK</span>`;

            else if (product.status === "Low Stock") statusHTML = `<span class="status-badge bg-yellow-500 text-black">LOW STOCK</span>`;

            else if (product.status === "In Transit") statusHTML = `<span class="status-badge bg-blue-500 text-white">IN TRANSIT</span>`;

            else if (product.status === "Coming Soon") statusHTML = `<span class="status-badge bg-amber-500 text-black">COMING SOON</span>`;

            card.innerHTML = `

                <div class="relative">

                    <img src="${product.image}" class="w-full h-56 object-cover">

                    <div class="absolute top-4 right-4">${statusHTML}</div>

                </div>

                <div class="p-6">

                    <div class="font-semibold text-lg">${product.name}</div>

                    <div class="text-xs text-slate-400">${product.brand} • ${product.mainCategory}</div>

                    <div class="flex justify-between items-center mt-4">

                        <div class="text-3xl font-bold">$${product.price}</div>

                        <button onclick="event.stopImmediatePropagation(); addToCart(${product.id});" class="px-5 py-2 bg-cyan-500 hover:bg-cyan-600 text-black font-bold text-sm rounded-2xl">ADD</button>

                    </div>

                </div>

            `;

            card.onclick = () => showProductModal(product);

            return card;

        }

        function addToCart(productId) {

            const product = products.find(p => p.id === productId);

            if (!product) return;

            const existing = cart.find(item => item.id === productId);

            if (existing) existing.quantity++;

            else cart.push({...product, quantity: 1});

            updateCartCount();

        }

        function updateCartCount() {

            const count = cart.reduce((sum, item) => sum + item.quantity, 0);

            document.getElementById('cart-count').textContent = count;

        }

        function showCart() {

            if (cart.length === 0) return alert("Cart is empty");

            let total = cart.reduce((sum, item) => sum + item.price * item.quantity, 0);

            showStripeCheckout(total);

        }

        function showStripeCheckout(total) {

            document.getElementById('stripe-total').innerText = '$' + total.toFixed(2);

            document.getElementById('stripe-modal').classList.remove('hidden');

            document.getElementById('stripe-modal').classList.add('flex');

        }

        function closeStripeModal() {

            document.getElementById('stripe-modal').classList.add('hidden');

            document.getElementById('stripe-modal').classList.remove('flex');

        }

        function startStripeCheckout() {

            const btn = event.currentTarget;

            btn.innerHTML = 'Processing...';

            btn.disabled = true;

                cart = [];

                updateCartCount();

        }
